\large \textbf{Rational Numbers}


\normalsize
Numbers that can be written in the form $a/b$, where $a$ and $b$ are both integers. An integer $n$ is also an integer because $n/1 = n$.
\begin{equation*}
    \frac{1}{2}, \quad \frac{11}{70}, \quad \frac{125}{7}, \quad -\frac{10}{11}, \quad 0=\frac{0}{1}, \quad -1.26=-\frac{126}{100}
\end{equation*}

There are no integers $p$ and $q$ such that $\sqrt{2} = p/q$. Hence, $\sqrt{2}$ is not rational.

\subsection{Integer Powers}

The expression $a^n$ is called the \textbf{nth power} of a
% \underbrace{} draws the curly brace under whatever in the braces
% _{} is the caption underneath
% \cdot is the centered multiplication dot
\begin{align*}
    &a^n = \underbrace{a \cdot a \cdot \ldots \cdot a}_{n \text{ factors}}\\[15pt]
    % \left( ~contents~ \right) scale the parentheses to the appropriate size
    % \hspace[-5pt] shifts the equation a bit left since the equation is not centered
    &\hspace{-19pt}\left(\frac{p}{q}\right)^5= \frac{p}{q} \cdot \frac{p}{q} \cdot \frac{p}{q} \cdot \frac{p}{q} \cdot \frac{p}{q}\\[-30pt]
    % add -30pt after the last equation to bring next paragraph closer
\end{align*}
$0^0$ is undefined. \\
In general, 
\begin{equation*}
    a^{-n}= \frac{1}{a^n} \quad \text{therefore,} \quad 3^{-2}= \frac{1}{9}
\end{equation*}
% neq is the symbol for not equal to 
whenever $n$ is a natural number and $a \neq 0$. \\
\large Properties of Powers \vspace{-15pt}


\normalsize 
\begin{equation*}
    \text{(i)} \; \; a^r \, \cdot \, a^s = a^{r+s} \quad \quad \text{(ii)} \; \; (a^r)^s = \; a^{rs}
\end{equation*}

Example 1: $a^3 \;  \cdot \;\ a^5 = \underbrace{a \, \cdot a \, \cdot a}_{3 \, \text{factors}} \, \cdot \; \underbrace{a \, \cdot a \, \cdot a \, \cdot a \, \cdot a}_{5 \, \text{factors}} = a^{3+5}=a^8$

% From the graphicx package used to scale the text down to one line
\scalebox{0.85}{Example 2: $(a^2)^4= \underbrace{a \, \cdot \, a}_{2 \, \text{factors}} \, \cdot \, \underbrace{a \, \cdot \, a}_{2 \, \text{factors}} \, \cdot \, \underbrace{a \, \cdot \, a}_{2 \, \text{factors}} \, \cdot \, \underbrace{a \, \cdot \, a}_{2 \, \text{factors}} = \underbrace{a \, \cdot \, a \, \cdot \, a \, \cdot \, a \, \cdot \, a \, \cdot \, a \, \cdot \, a \cdot \, a}_{8 \, \text{factors}} = a^{3 + 5}=a^8$ \hspace{-20pt}} \vspace{7pt}

Same base division:
\begin{equation*}
    a^r \, \div \, a^s=\frac{a^r}{a^s}=a^r\frac{1}{a^s}=a^r \, \cdot \, a^{-s}=a^{r-s}\\
\end{equation*}
\vspace{-60pt}
\begin{equation*}
    a^3 \div a^5=a^{3-5}=a^{-2}
\end{equation*}
\vspace{-20pt}
\begin{equation*}
    (ab)^r = \underbrace{ab \, \cdot ab \, \cdot \ldots \cdot \, ab}_{r \, \text{factors}} = \underbrace{a \, \cdot a \, \cdot \ldots \cdot \, a}_{r \, \text{factors}} \, \cdot \, \underbrace{b \, \cdot b \, \cdot \ldots \cdot \, b}_{r \, \text{factors}} = a^rb^r
\end{equation*}
\vspace{-15pt}

% overbrace function draws brace above contents, uses ^ instead of _ for the label
\begin{equation*}
    \left( \frac{a}{b} \right)^r = \underbrace{\frac{a}{b} \, \cdot \, \frac{a}{b} \, \cdot \, \ldots \, \cdot \frac{a}{b}}_{r \, \text{factors}} = \frac{\overbrace{a \, \cdot a \, \cdot \ldots \, \cdot a}^{r \, \text{factors}}}{\underbrace{b \, \cdot b \, \cdot \ldots \, \cdot b}_{r \, \text{factors}}}=\frac{a^r}{b^r}=a^rb^{-r}
\end{equation*}

\large Compound Interest
\normalsize

A quantity K which increases by p\% per year will have increased after t years to
\begin{equation*}
    K \left(1+\frac{p}{100}\right)^t
\end{equation*}
Here $1 + \frac{p}{100}$ is the growth factor for a growth of p\%. $(1.08)^t$ is the amount to which 1\$ has increased over $t$ years at 8\% interest. The same applies to depreciating items. A car bought for \$15,000 dollars depreciates by 15\% per year over a six-year period. It's value over 6 years is:
% is approximate to symbol is \approx
\begin{equation*}
    \$15,000\left( 1- \frac{p}{100} \right)^6 \approx \$5,657
\end{equation*}
